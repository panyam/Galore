JSON Parser

This example shows how to build a complete JSON parser with Galore.

Grammar Definition

Value   -> Object | Array | STRING | NUMBER | "true" | "false" | "null"
Object  -> "{" "}" | "{" Members "}"
Members -> Member | Members "," Member
Member  -> STRING ":" Value
Array   -> "[" "]" | "[" Elements "]"
Elements -> Value | Elements "," Value

STRING  -> /"[^"]*"/
NUMBER  -> /-?[0-9]+(\.[0-9]+)?([eE][+-]?[0-9]+)?/

Implementation

import * as G from "galore";

const jsonGrammar = `
Value   -> Object | Array | STRING | NUMBER | "true" | "false" | "null"
Object  -> "{" "}" | "{" Members "}"
Members -> Member | Members "," Member
Member  -> STRING ":" Value
Array   -> "[" "]" | "[" Elements "]"
Elements -> Value | Elements "," Value

STRING  -> /"[^"]*"/
NUMBER  -> /-?[0-9]+(\.[0-9]+)?([eE][+-]?[0-9]+)?/
`;

const parser = G.newParser(jsonGrammar, { type: "lalr" });

// Parse JSON string
const result = parser.parse('{"name": "Galore", "version": 1.0}');
console.log(result);
